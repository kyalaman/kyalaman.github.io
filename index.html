      function findPrimes(n) {
        var primes = new Array(n+1).fill(true);
        var max = Math.sqrt(n);

        for (i = 2; i <= max; i++) {
          if (primes[i]) {
            for (j = i*i; j <= n; j += i) {
              primes[j] = false;
            }
          }
        }
        var list = [];
        for (k = 2; k <= n; k++) {
          if (primes[k]) {
            list.push(k);
          }
        }
        return list;
      }
